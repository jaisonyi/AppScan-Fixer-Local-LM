JavaScript · MCP Client Core (mcpClient.js) 
// ─────────────────────────────────────────────────────
// AppScan360 MCP Client  ·  mcpClient.js
// Implements MCP HTTP transport for AppScan360
// ─────────────────────────────────────────────────────

class AppScan360MCPClient {
  constructor(config) {
    this.url    = config.url    // e.g. "https://as360aio.appscan.local/mcp"
    this.apiKey = config.apiKey // "keyId:keySecret"
    this.tools  = []
    this.sessionId = null
  }

  // Build auth header from composite key
  _headers() {
    return {
      'Content-Type' : 'application/json',
      'X-API-KEY'    : this.apiKey,
      'Accept'       : 'application/json'
    }
  }

  // MCP initialize handshake
  async initialize() {
    const res = await fetch(this.url, {
      method : 'POST',
      headers: this._headers(),
      body   : JSON.stringify({
        jsonrpc: '2.0', id: 1,
        method : 'initialize',
        params : {
          protocolVersion: '2024-11-05',
          capabilities   : { tools: {} },
          clientInfo     : { name: 'ollama-mcp-client', version: '1.0.0' }
        }
      })
    })
    const data = await res.json()
    if (data.error) throw new Error(data.error.message)
    this.sessionId = data.result?.sessionId
    return data.result
  }

  // Discover tools available on the MCP server
  async listTools() {
    const res = await fetch(this.url, {
      method : 'POST',
      headers: this._headers(),
      body   : JSON.stringify({
        jsonrpc: '2.0', id: 2,
        method : 'tools/list', params: {}
      })
    })
    const data = await res.json()
    this.tools = data.result?.tools || []
    return this.tools
  }

  // Invoke a named MCP tool with given arguments
  async callTool(name, args = {}) {
    const res = await fetch(this.url, {
      method : 'POST',
      headers: this._headers(),
      body   : JSON.stringify({
        jsonrpc: '2.0', id: Date.now(),
        method : 'tools/call',
        params : { name, arguments: args }
      })
    })
    const data = await res.json()
    if (data.error) throw new Error(data.error.message)
    return data.result?.content
  }

  // Convenience: get all scans
  async getScans(filter = {}) {
    return this.callTool('get_scans', filter)
  }

  // Convenience: get issues for a scan or all
  async getIssues(params = {}) {
    return this.callTool('get_issues', params)
  }

  // Convenience: get apps/applications
  async getApplications() {
    return this.callTool('get_applications', {})
  }
}

export default AppScan360MCPClient
JavaScript · Ollama LLM Bridge (ollamaBridge.js) 
// ─────────────────────────────────────────────────────
// Ollama LLM Bridge  ·  ollamaBridge.js
// Streams chat completions from a local Ollama instance
// ─────────────────────────────────────────────────────

class OllamaBridge {
  constructor(baseUrl = 'http://localhost:11434', model = 'llama3.1') {
    this.baseUrl = baseUrl
    this.model   = model
    this.system  = this._buildSystemPrompt()
  }

  _buildSystemPrompt() {
    return `You are an application security analyst assistant integrated with
HCL AppScan360 via the Model Context Protocol (MCP). You have access
to real-time scan data, vulnerability findings, and application
inventory from the AppScan360 platform.

When analyzing security data:
- Prioritize Critical and High severity issues for immediate action
- Map findings to CWE categories and OWASP Top 10 where applicable
- Provide actionable remediation guidance with code examples
  - Use global baseline mappings by default: OWASP ASVS, NIST SSDF, ISO/IEC 27001, and CIS Controls
  - Add industry or regulatory mappings when context indicates relevance: PCI DSS (payment), GDPR/CCPA (privacy), SOC 2 (assurance)
  - Include Korean KISA SDLC guidelines and 행안부 고시 only when Korea jurisdiction or customer context is explicitly relevant
  - Do not default to a single regional framework unless explicitly requested
- If the user requests a specific framework, prioritize it first and provide a short crosswalk to at least one global baseline
- State "Not enough evidence" when data does not support a compliance claim
- Use this response structure for compliance analysis:
  1) Global Baseline Mapping (OWASP ASVS / NIST SSDF / ISO 27001 / CIS)
  2) Industry/Regulatory Add-ons (PCI DSS, GDPR/CCPA, SOC 2) when applicable
  3) Regional Add-ons (KISA etc.) only when explicitly relevant
  4) Evidence and Gaps (finding -> control mapping, pass/fail/partial, rationale)
  5) Prioritized Remediation Plan (top actions with owner and target timeline)
- Reference CVSS scores when assessing business risk
- Be concise but technically precise in your assessments`
  }

  // List available models from local Ollama
  async listModels() {
    const res  = await fetch(`${this.baseUrl}/api/tags`)
    const data = await res.json()
    return (data.models || []).map(m => m.name)
  }

  // Non-streaming chat completion
  async chat(messages, options = {}) {
    const payload = {
      model   : options.model || this.model,
      messages: [
        { role: 'system', content: this.system },
        ...messages
      ],
      stream  : false,
      options : {
        num_ctx : options.numCtx   || 8192,
        temperature : options.temp || 0.3
      }
    }
    const res  = await fetch(`${this.baseUrl}/api/chat`, {
      method : 'POST',
      headers: { 'Content-Type': 'application/json' },
      body   : JSON.stringify(payload)
    })
    const data = await res.json()
    return data.message?.content || ''
  }

  // Streaming chat completion — calls onChunk(text) for each token
  async chatStream(messages, onChunk, options = {}) {
    const res = await fetch(`${this.baseUrl}/api/chat`, {
      method : 'POST',
      headers: { 'Content-Type': 'application/json' },
      body   : JSON.stringify({
        model   : options.model || this.model,
        messages: [{ role: 'system', content: this.system }, ...messages],
        stream  : true,
        options : { num_ctx: options.numCtx || 8192, temperature: 0.3 }
      })
    })
    const reader  = res.body.getReader()
    const decoder = new TextDecoder()
    let full = ''
    while (true) {
      const { done, value } = await reader.read()
      if (done) break
      const lines = decoder.decode(value).split('\n').filter(Boolean)
      for (const line of lines) {
        try {
          const chunk = JSON.parse(line)
          const text  = chunk.message?.content || ''
          full += text
          onChunk(text, full)
        } catch {}
      }
    }
    return full
  }
}

export default OllamaBridge
JavaScript · Analysis Pipeline (pipeline.js) 
// ─────────────────────────────────────────────────────
// Analysis Pipeline  ·  pipeline.js
// Orchestrates MCP data retrieval + Ollama LLM analysis
// ─────────────────────────────────────────────────────

import AppScan360MCPClient from './mcpClient.js'
import OllamaBridge       from './ollamaBridge.js'

class AppSecPipeline {
  constructor({ mcpUrl, apiKeyId, apiKeySecret, ollamaUrl, model }) {
    this.mcp    = new AppScan360MCPClient({
      url   : mcpUrl,
      apiKey: `${apiKeyId}:${apiKeySecret}`
    })
    this.llm    = new OllamaBridge(ollamaUrl, model)
    this.history = []
  }

  async init() {
    await this.mcp.initialize()
    await this.mcp.listTools()
    return this.mcp.tools
  }

  // Fetch security context from AppScan360 then ask Ollama
  async query(userMessage, onChunk) {
    // 1. Retrieve relevant context from MCP
    let context = ''
    try {
      const scans  = await this.mcp.getScans()
      const issues = await this.mcp.getIssues({ limit: 50 })
      context = `## AppScan360 Live Data\n\n
### Recent Scans\n${JSON.stringify(scans, null, 2)}

### Recent Issues (top 50)\n${JSON.stringify(issues, null, 2)}`
    } catch (e) {
      context = `(Unable to retrieve live data: ${e.message})`
    }

    // 2. Build conversation with injected context
    const augmentedUser = `${context}

---
User question: ${userMessage}`

    this.history.push({ role: 'user', content: augmentedUser })

    // 3. Stream response from Ollama
    let reply = ''
    if (onChunk) {
      reply = await this.llm.chatStream(this.history, onChunk)
    } else {
      reply = await this.llm.chat(this.history)
    }

    this.history.push({ role: 'assistant', content: reply })
    return reply
  }

  // Dedicated: triage all open issues and produce a severity report
  async triageIssues() {
    const issues = await this.mcp.getIssues({ status: 'open' })
    return this.llm.chat([{
      role: 'user',
      content: `Perform a full triage of the following open AppScan360 issues.
Group by severity, identify the top 3 immediate actions, and provide
CWE mappings. Data:\n\n${JSON.stringify(issues, null, 2)}`
    }])
  }

  // Dedicated: generate a compliance gap report (KISA / OWASP)
  async complianceReport(framework = 'OWASP ASVS + NIST SSDF + ISO/IEC 27001') {
    const issues = await this.mcp.getIssues({})
    return this.llm.chat([{
      role: 'user',
      content: `Map the following AppScan360 findings to the ${framework} framework.
Identify compliance gaps, reference specific control IDs, and
produce a pass/fail summary table. Data:\n\n${JSON.stringify(issues, null, 2)}`
    }])
  }
}

export default AppSecPipeline

Python · Alternative CLI Client (appscan_ollama.py) 
#!/usr/bin/env python3
# ─────────────────────────────────────────────────────────
# AppScan360 MCP + Ollama CLI Client  ·  appscan_ollama.py
# Requires: pip install requests ollama
# ─────────────────────────────────────────────────────────

import json, requests, ollama, sys
from datetime import datetime

MCP_URL        = "https://as360aio.appscan.local/mcp"
API_KEY_ID     = "local_635907e4-b313-527c-1d94-2ba8d495ec1d"
API_KEY_SECRET = "zJqWHnyymqZAqDvdUogw2AQm/LNIppcTwZVeX1DpqZHg"
OLLAMA_MODEL   = "llama3.1:8b"  # or "mistral", "codellama", etc.

SYSTEM_PROMPT = """You are an application security analyst integrated with
HCL AppScan360. Analyze vulnerability data, map to CWE and global security standards
(OWASP ASVS, NIST SSDF, ISO/IEC 27001, CIS Controls), and include region/industry frameworks
only when context requires them (e.g., KISA for Korea, PCI DSS for payments, GDPR/CCPA for privacy).
Provide actionable remediation guidance. Be concise and precise."""


def mcp_call(method: str, params: dict = {}) -> dict:
    """Send a JSON-RPC 2.0 request to the AppScan360 MCP endpoint."""
    headers = {
        "Content-Type" : "application/json",
        "X-API-KEY"     : f"{API_KEY_ID}:{API_KEY_SECRET}",
    }
    payload = {"jsonrpc": "2.0", "id": 1, "method": method, "params": params}
    r = requests.post(MCP_URL, json=payload, headers=headers, verify=False, timeout=30)
    r.raise_for_status()
    result = r.json()
    if "error" in result:
        raise RuntimeError(result["error"]["message"])
    return result.get("result", {})


def call_tool(name: str, args: dict = {}) -> list:
    """Invoke a named MCP tool and return the content array."""
    result = mcp_call("tools/call", {"name": name, "arguments": args})
    return result.get("content", [])


def get_context() -> str:
    """Pull live scan and issue data to inject into LLM context."""
    scans  = call_tool("get_scans")
    issues = call_tool("get_issues", {"limit": 50})
    return f"""## AppScan360 Live Context  ({datetime.now().strftime('%Y-%m-%d %H:%M')})

### Scans
{json.dumps(scans, indent=2)}

### Issues (top 50)
{json.dumps(issues, indent=2)}"""


def chat_with_context(user_msg: str, history: list) -> str:
    """Fetch MCP context, augment the user message, stream from Ollama."""
    print("  [mcp] fetching context…", end="", flush=True)
    context = get_context()
    print(" done")

    augmented = f"""{context}

---
User: {user_msg}"""

    messages = [{"role": "system", "content": SYSTEM_PROMPT}]
    messages += history
    messages.append({"role": "user", "content": augmented})

    reply = ""
    print("\n[assistant] ", end="", flush=True)
    for chunk in ollama.chat(model=OLLAMA_MODEL, messages=messages, stream=True):
        token = chunk["message"]["content"]
        reply += token
        print(token, end="", flush=True)
    print()
    return reply


def main():
    print("Connecting to AppScan360 MCP…")
    mcp_call("initialize", {
        "protocolVersion": "2024-11-05",
        "capabilities"   : {"tools": {}},
        "clientInfo"     : {"name": "ollama-cli", "version": "1.0"}
    })
    tools = mcp_call("tools/list")
    print(f"  Connected. {len(tools.get('tools', []))} tools available.")
    print(f"  Ollama model: {OLLAMA_MODEL}\n")

    history = []
    while True:
        try:
            user_input = input("[you] ").strip()
        except (EOFError, KeyboardInterrupt):
            print("\nExiting."); sys.exit(0)
        if not user_input: continue
        if user_input.lower() in ("exit", "quit"): break
        reply = chat_with_context(user_input, history)
        history.append({"role": "user",      "content": user_input})
        history.append({"role": "assistant", "content": reply})


if __name__ == "__main__":
    main()
